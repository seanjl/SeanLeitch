Java Triangle Exercise

6 Oct 2019

This is a java app which receives 3 inputs (sides of a triangle) and outputs what type of triangle it is

The user inputs the length of each side of the triangle.

package uem.dam.m11.tareaejercicios; import java.util.Scanner; public class EjercicioTriangulo { public static void main(String[] args) { // Mensaje que dice al usuario "Introduce los lados del triangulo:" System.out.println("Introduce los lados del triangulo"); // Lee los lados del triangulo Scanner sc = new Scanner(System.in); int lado1 = sc.nextInt(); int lado2 = sc.nextInt(); int lado3 = sc.nextInt(); // Mensaje que dice al usuario "El tri�ngulo es:" System.out.println("El tri�ngulo es: "); /* Operador ternario / Condicional: Compara los lados del triangulo * e indica al usuario que tipo de tri�ngulo es. */ if(lado1==lado2 && lado2==lado3) System.out.println("Equilatero"); else if ((lado1==lado2 && lado2!=lado3 ) || (lado1!=lado2 && lado3==lado1) || (lado3==lado2 && lado3!=lado1)) System.out.println("Isosceles"); else if(lado1!=lado2 && lado2!=lado3 && lado3!=lado1) System.out.println("Escaleno"); sc.close(); } }
